CREATE TABLE item(
id SERIAL PRIMARY KEY,
isactive BOOL DEFAULT TRUE,
title VARCHAR(100) NOT NULL,
img VARCHAR ARRAY NOT NULL,
text TEXT NOT NULL,
price INT NOT NULL,
prew_img TEXT NOT NULL,
creator_id INT NOT NULL,
ad TEXT,
lat TEXT,
log TEXT,
k TEXT
);

CREATE TABLE users(
id SERIAL PRIMARY KEY,
email VARCHAR(100) UNIQUE NOT NULL,
password VARCHAR(100) NOT NULL,
name VARCHAR(100) UNIQUE NOT NULL,
prew_img TEXT NOT NULL,
main_num VARCHAR(11) UNIQUE NOT NULL
);
CREATE TABLE comment(
id SERIAL PRIMARY KEY,
post_id INT NOT NULL,
mess TEXT NOT NULL,
pub_date DATE NOT NULL,
name TEXT NOT NULL
);

CREATE TABLE chat(
id SERIAL PRIMARY KEY,
us_2 TEXT,
us_1 TEXT,
mess TEXT ARRAY,
pub_date DATE ARRAY,
name TEXT ARRAY
);
